# Makefile

```shell
#$@  表示目标文件
#$^  表示所有的依赖文件
#$<  表示所有的依赖文件中的第一个文件
test : test.go
	go build -o $@ $^
```

```makefile
test:test.o test1.o
    gcc -o $@ $^
    
#"%.o" 把我们需要的所有的 ".o" 文件组合成为一个列表，
#从列表中挨个取出的每一个文件，"%" 表示取出来文件的文件名（不包含后缀），
#然后找到文件中和 "%"名称相同的 ".c" 文件，然后执行下面的命令，直到列表中的文件全部被取出来为止。
%.o:%.c
    gcc -o $@ $^
```
